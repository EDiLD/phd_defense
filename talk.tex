\begin{frame}
\frametitle{Analysing chemical concentrations}
	\begin{itemize}
		\pause
		\item Concentrations \textless~LOQ (96\% of all measurements)
		\pause
		\item Hurdle-model: 
			\begin{align*}
			y \sim ZAGA = 
			  \begin{cases}
			    Binomial~GLM   & \quad  \text{if } y < LOQ \\
			    Gamma~GLM & \quad \text{if } y \ge LOQ \\
			  \end{cases}
			\end{align*}
		\pause
		\item Risk Quotient
			\begin{itemize}
				\item \large $RQ = \frac{c}{RAC}$
			\end{itemize}
		\pause
		\item Predictors
		    \begin{itemize}
				\item Catchment size 
				\item Agricultural land use
				\item Precipitation
			\end{itemize}
	\end{itemize}
\end{frame}

	\begin{columns}[T]

	    	\begin{itemize}
	        	\item factorial design
	        	\item one predictor
	        	\item low replicated
	        	\item LM, GLM, bootstrap

	        	\item continuous design
	        	\item two predictors
	        	\item well replicated
	        	\item LM, GLM
